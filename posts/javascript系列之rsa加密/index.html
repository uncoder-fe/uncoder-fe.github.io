没错，拷贝就可以用了，注意下\r

使用的库

JSEncrypt;

代码

getRsa:function(str) {
    var jse = new JSEncrypt();
    var pubKey ="-----BEGIN PUBLIC KEY-----\r-----END PUBLIC KEY-----";
    jse.setPublicKey(pubKey);
    var encrypted = jse.encrypt(str);
    return encrypted;
}

备注：/r换行
